add_namespace = spain

#Boom ends
country_event = {
		id = spain.1
		title = spain.1.t
		desc = spain.1.d
		picture = GFX_report_event_generic_factory
		
		fire_only_once = yes
		
		trigger = {
			TAG = SPR
			has_idea = SPR_Boom
			date > 1921.6.1
		}
		
		option = {
			name = spain.1.a
			remove_ideas = SPR_Boom
		}
}

#Dictatorship starts
country_event = {
		id = spain.2
		title = spain.2.t
		desc = spain.2.d
		
		fire_only_once = yes
		
		trigger = {
			TAG = SPR
			date > 1923.1.1
		}
		
		option = {
			name = spain.2.a
			add_ideas = SPR_PrimoDictatorship
			set_politics = { ruling_party = neutrality elections_allowed = no }
			add_popularity = { ideology = neutrality popularity = 0.25 }
		}
}

#Dictatorship ends
country_event = {
		id = spain.3
		title = spain.3.t
		desc = spain.3.d
		
		fire_only_once = yes
		
		trigger = {
			TAG = SPR
			date > 1930.1.1
			has_idea = SPR_PrimoDictatorship
		}
		
		option = {
			name = spain.3.a
			remove_ideas = SPR_PrimoDictatorship
		}
}

#Spain turns democratic
country_event = {
		id = spain.4
		title = spain.4.t
		desc = spain.4.d
		
		fire_only_once = yes
		
		trigger = {
			TAG = SPR
			date > 1931.6.1
			NOT = { SPR = { has_government = democratic } }
		}
		
		option = {
			name = spain.4.a
			set_politics = { ruling_party = democratic elections_allowed = yes }
			add_popularity = { ideology = democratic popularity = 0.25 }
		}
}
